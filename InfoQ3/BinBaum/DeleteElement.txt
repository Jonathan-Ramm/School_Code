string deleteElement(int Nummer){
    int art = whatKind(Nummer);
    int Nr;
    string was;
    string inhalt="nicht vorhanden\n";
    if(art > -1 ){
        inhalt = runner->NN;
        if( art ==2){ // egal ob Wurzel oder nicht
            guard = runner->left;
            change = runner;
            while(guard->right!=0){
                guard = guard->right;
            }// steht auf dem Austauschelement
            Nr = guard->No;
            was= guard->NN;
            deleteElement(Nr);
            change->NN = was;
            change->No = Nr;
        }else{
            if(runner != root){ // nicht die Wurzel
                guard = runner->back;//Einarm oder Blatt
                if (art == 0){
                    if(guard->left == runner){delete runner;guard->left=0; } //  Blatt h채ngt links
                    if(guard->right == runner){delete runner;guard->right=0;} // Blatt h채ngt rechts
                }
                if (art == 1){
                    if(guard->left == runner){ //h채ngt links
                        if(runner->left !=0){guard->left = runner->left;delete runner; // hat links  ein Kind
                        }else{guard->left = runner->right;delete runner;} // hat rechts ein Kins
                    }
                    if(guard->right == runner){//h채ngt rechts
                        if(runner->left !=0){guard->right = runner->left;delete runner;// hat links  ein Kind
                        }else{guard->right = runner->right;delete runner;}// hat rechts  ein Kind
                    }
                }
            }else{
                if(art ==0){ // die Wurzel
                    delete root; // Baum hat nur die Wurzel
                }else{
                    if(runner->left!=0){
                        guard = root; root = root->left; delete guard; //hat links ein Kind
                    }
                    if(runner->right!=0){
                        guard = root; root = root->right; delete guard;// hat rechts  ein Kind
                    }
                }
            }
        }
    }
    return inhalt;
}
